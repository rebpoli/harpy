\subsection{Mechanics}
The Cauchy total stress is defined as:
\begin{align}
    \ssigma = \CC : \eeps - \alpha\ P\ \II - \alpha_d\ \II\ T
\end{align}
where $\CC$ is the elastic stiffness tensor, $\eeps=\frac{1}{2} ( \nabla \uu + \nabla^T \uu )$ is the strain tensor, 
$\uu$ is the displacement vector, $\alpha$ is the Biot coefficient, $\alpha_d=K\beta_d$ is the 
drained thermoelastic effective stress coefficient, $K$ is the drained bulk moduluis,
$\beta_d=3\beta_l$ is the drained coefficient
of volumetric thermal expansion, $\beta_l$ is the coefficient of linear thermal expansion, $P$ is
the pore pressure and $T$ is the temperature.

The mechanical equilibrium states that:
\begin{align}
  \nabla \cdot \ssigma + \bb = 0
\end{align}
where $\bb$ is the body force, typically associated to the gravity acceleration.

\subsection{Heat transport - thermal}
\begin{align}
  \alpha_d\ \Tzero\ \dot{\epsilon} - \beta_e\ \Tzero\ \dot{P} + (\rho c)_m\ \dot{T} = \nabla\cdot(\kappa_m\ \nabla T)
\end{align}
where $\epsilon=\varepsilon_{kk}$ is the volumetric strain, $\beta_e$ is the coefficient of volumetric thermal expansion for variation in fluid
content at constant frame volume, $\Tzero$ is the initial temperature, assuming 
\[|T-\Tzero|/\Tzero\ll1 \qquad ,\]
and defining 
\[(\rho c)_m=(1-\phi) (\rho c)_s + \phi (\rho c)_f \qquad,\] 
\[\kappa_m = (1-\phi) \kappa_s + \phi \kappa_f \qquad,\]
$\rho$ is the density of the fluid $f$, solid $s$
or the mixture $m$, and $c$ is the specific heat of each phase, $\kappa$ is the thermal conductivity
of each phase, $\phi$ is the porosity.

%
%
%
\subsection{Hydraulic flow}
\begin{align}
  \alpha\ \dot{\epsilon} + \frac{1}{M}\ \dot{P} - \beta_e\ \dot{T} = \nabla\cdot( k\ \nabla P )
\end{align}
where  $M$ is the Biot modulus and $k=\hat{k}/\mu$ is the 
hydraulic conductivity, $\hat{k}$ is the intrinsic permeability and $\mu$ is the fluid viscosity 
(assuming single phase).
